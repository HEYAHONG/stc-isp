3. shell script

you can add this two func to your ~/.bashrc, One key download:

1). one key compile, will makedir build in current dir.

stc-cc() {
CUR_PATH=`pwd`
    if [ ! -d $CUR_PATH/build ]; then
        mkdir $CUR_PATH/build
    fi

    cd $CUR_PATH/build
    sdcc -mmcs51 $CUR_PATH/$1
}

2). one key compile and download to the chip!

stc-cisp() {
CUR_PATH=`pwd`
# make build in current dir
    if [ ! -d $CUR_PATH/build ]; then
        mkdir $CUR_PATH/build
    fi

# compile
    sourceFile=$CUR_PATH/build/$1
    cd $CUR_PATH/build
    sdcc -mmcs51 $CUR_PATH/$1
    cd $CUR_PATH
    stcisp -f ${sourceFile%.*}.ihx
}
