#### 1). 一键编译, 完成后在当前目录下生成build文件夹
stc-cc() {
CUR_PATH=`pwd`
    if [ ! -d $CUR_PATH/build ]; then
        mkdir $CUR_PATH/build
    fi

    cd $CUR_PATH/build
    sdcc -mmcs51 $CUR_PATH/$1
}

2). 一键编译&&烧写

stc-cisp() {
CUR_PATH=`pwd`
# 首先在当前目录下生成build目录
    if [ ! -d $CUR_PATH/build ]; then
        mkdir $CUR_PATH/build
    fi

# 编译
    sourceFile=$CUR_PATH/build/$1
    cd $CUR_PATH/build
    sdcc -mmcs51 $CUR_PATH/$1
    cd $CUR_PATH
    stcisp -f ${sourceFile%.*}.ihx
}
